for문

for 문은 괄호로 감싸고 세미콜론으로 구분한 세 개의 선택식과, 반복을 수행할 문(주로 블럭문)으로 이루어져 있습니다.

for (초기 값; 조건식; 증가감식;){
    실행문;
}

1부터 100까지 출력하기

//1부터 100까지 출력하기
for (let i = 1; i <= 100; i++){ // i = i + 1 와 같다.
    document.write(i);
}
// 결과는 : 1 2 3 4 5﹒﹒﹒﹒﹒100

1부터 100까지 짝수 출력하기

짝수로 출력할 때는 증가값을 2로 해주면 됩니다.

//1부터 100까지 짝수 출력하기
for (let i = 2; i <= 100; i+=2){ // i = i + 2 와 같다.
    document.write(i);
}
// 결과는 : 2 4 6 8﹒﹒﹒﹒﹒100

for문을 이용해서 테이블 출력하기

중첩 for문을 이용해서 테이블을 만들 수 있습니다.

//중첩 for문을 이용해서 테이블 만들기
let table = "<table>";
// table += <tr>;
// table += <td>;
// table += </td>;
// table += </tr>;
// table += </table>;
for(let i=1; i<=9; i++){
    table += "<tr>";
                
    for(let j=1; j<=9; j++){
        table += "<td>" + i + " * " + j + " = " + i*j + "</td>";
    }
    table += "</tr>";
}
    table += "</table>";
    document.write(table);
